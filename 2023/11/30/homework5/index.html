第九章第5题

\[ \begin{aligned} \\&任取a,b\in G， \\&\left(1\right)证充分性： \\&由群同态定义，\phi (ab)=\phi (a)\phi (b) \\&则abab=aabb \\&则ba=ab，满足交换律， \\&\phi 是一种群同态时G为阿贝尔群，充分性得证 \\&\left(2\right)证必要性： \\&G为阿贝尔群,则G中元素满足交换律 \\&则\phi (ab)=(ab)^2=abab=aabb=\phi (a)\phi (b) \\&对任意a,b\in G，群操作得以保持， \\&G为阿贝尔群时\phi 是一种群同态，必要性得证 \end{aligned} \]

第九章第6题

\[ \begin{aligned} \\&\left(一\right)证G是循环群时\phi (G)也是循环群： \\&设g为G的生成元，g在H的映射为\phi (g) \\&\phi (G)=<\phi (g)>时，\phi (G)是循环群 \\&需证\phi (G)=<\phi (g)> \\&\left(1\right)证\phi (G)\subset <\phi (g)>： \\&任取\phi (g')\in \phi (G)，则g'\in G \\&因为G为循环群，存在i，g'=g^i \\&所以\phi (g')=\phi (g^i)=(\phi (g))^{i}\in <\phi (g)> \\&所以\phi (G)\subset <\phi (g)> \\&\left(1\right)证<\phi (g)>\subset \phi (G)： \\&任取(\phi (g))^{i}\in <\phi (g)>，(\phi (g))^{i}=\phi (g^i) \\&因为G为循环群，g^i\in G, \phi (g^i)\in \phi (G) \\&(\phi (g))^{i}\in \phi (G),所以<\phi (g)>\subset \phi (G) \\&所以\phi (G)=<\phi (g)>，G是循环群时\phi (G)也是循环群 \\&\left(二\right)证G是交换群时\phi (G)也是交换群： \\&因为G是交换群，任取g_1,g_2\in G, g_1g_2=g_2g_1 \\&\phi (g_1)\phi (g_2)=\phi (g_1g_2)=\phi (g_2g_1)=\phi (g_2)\phi (g_1) \\&\phi (G)满足交换律，为交换群 \\&所以G是交换群时\phi (G)也是交换群 \end{aligned} \]

第九章第7题

\[ \begin{aligned} \\&因为[G:H]=2，所以H在G上有两个左陪集，令与H不同的陪集为K \\&\left(1\right)g\in H： \\&由群公理的封闭性，任取h\in H,gh\in H,hg\in H \\&则gH=Hg=H \\&\left(2\right)g\in K： \\&g\not\in H，gH\ne H，Hg\ne H \\&则由陪集的同一或不相交性，gH=Hg=G-H \\&所以，gH=Hg \end{aligned} \]

第九章第8题

\[ \begin{aligned} \\&任取g_1,g_2\in G,则g_2H,g_2H\in G/H \\&因为H是群G的正规子群，则g_1H=Hg_1,g_2H=Hg_2 \\&又因为G为阿贝尔群,则g_1,g_2满足交换律 \\&所以g_1Hg_2H=g_1g_2HH=g_2g_1HH=g_2Hg_1H \\&商群G/H满足交换律，所以商群G/H也是阿贝尔群 \end{aligned} \]

第九章第9题

\[ \begin{aligned} \\&因为G是循环群，所以G=<g>,若G/H也是循环群，则G/H=<gH> \\&需证G/H=<gH> \\&\left(1\right)证G/H\subset <gH>： \\&G/H=\{g_1H,g_2H,...,g_nH\} \\&对任意g_iH\in G/H,g_iH\in <gH> \\&G/H\subset <gH> \\&\left(2\right)证<gH>\subset G/H： \\&(gH)^i=g^iH,所以<gH>=\{g^1H,g^2H,...,g^nH\} \\&对任意g^iH\in <gH>, \\&因为G为循环群，G=\{g^1,g^2,...,g^n\} \\&所以g^i\in G \\&所以g^iH\in G/H \\&可得<gH>\subset G/H \\&所以，G/H=<gH>，得证 \end{aligned} \]
